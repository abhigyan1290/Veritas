    latency_ms REAL NOT NULL,
    cost_usd REAL NOT NULL,
    code_version TEXT,
    timestamp TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'ok',
    estimated INTEGER NOT NULL DEFAULT 0
);
